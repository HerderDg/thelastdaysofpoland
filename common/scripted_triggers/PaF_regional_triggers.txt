		has_global_flag = needed_arrays_initialized

			value = s
			OR = {
				ROOT = {
					owns_state = s
				}
				any_of_scopes = {
					array = ROOT.allies
					this = {
						owns_state = s
					}
				}
				any_of_scopes = {
					array = ROOT.subjects
					this = {

	}
