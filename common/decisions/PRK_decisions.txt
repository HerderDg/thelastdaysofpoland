ziobro_war = {
	
	ziobro_holy_war = {

		allowed = {
			always = no
		}

		available = {
			has_war = no
			NOT = {
				AND = {
					country_exists = STA
					country_exists = SND
				}
			}
		}

		icon = mission_icon   
		fire_only_once = yes
		days_mission_timeout = 90
		complete_effect = {
			log = "[GetDateText]: Decision ziobro_holy_war win"
			clr_country_flag = ziobro_start_war
			set_country_flag = ziobro_start_war_victory
			every_country = {
				limit = { NOT = { is_ai = yes } }
				news_event = { id = ziobro.22 }
			}
		}
		timeout_effect = {
			log = "[GetDateText]: Decision ziobro_holy_war lose"
			if = { # to jest wygrana
				limit = { 
					STA = { num_of_controlled_states < 1 }
					SND = { num_of_controlled_states < 1 }	
				}	
				clr_country_flag = ziobro_start_war
				set_country_flag = ziobro_start_war_victory
				every_country = {
					limit = {
						NOT = { is_ai = yes }
					}
					news_event = { id = ziobro.1 }
				}
				every_country = {
					limit = {
						has_war_with = PRK
					}
					add_to_variable = {
						global.deadPolacksFromPastWars = this.casualties_k
					}
					white_peace = PRK
				}					
				PRK = {
					add_to_variable = {
						global.deadPolacksFromPastWars = this.casualties_k
					}
					end_puppet = OSW
					end_puppet = CHW
				}
				OSW = { drop_cosmetic_tag = yes }				
			}
			else = { # to jest stalemate
				clr_country_flag = ziobro_start_war
				set_country_flag = ziobro_start_war_lose
				SND = {
					add_to_variable = {
						global.deadPolacksFromPastWars = this.casualties_k
					}
				}
				STA = {
					add_to_variable = {
						global.deadPolacksFromPastWars = this.casualties_k
					}
				}
				PRK = {
					country_event = { 
						id = ziobro.17
					}
					ROOT = {
						add_to_variable = {
							global.deadPolacksFromPastWars = this.casualties_k
						}
					}
					CHW = { drop_cosmetic_tag = yes }
					OSW = { drop_cosmetic_tag = yes }
				}
			}
		}
	}
