ziobro_war = {
	
	ziobro_holy_war = {

		allowed = {
			always = no
		}

		available = {
			date < 2027.06.01
			has_war = no
			NOT = {
				AND = {
					country_exists = STA
					country_exists = SND
					}
				}
			}

		icon = mission_icon   
		fire_only_once = yes
		days_mission_timeout = 90
		complete_effect = {
			log = "[GetDateText]: Decision ziobro_holy_war win"
			clr_country_flag = ziobro_start_war
			set_country_flag = ziobro_start_war_victory
			every_country = {
				limit = { NOT = { is_ai = yes } }
				news_event = { id = ziobro.1 }
				}
		}
		timeout_effect = {
			log = "[GetDateText]: Decision ziobro_holy_war lose"
			if = {
				limit = { 
					OR = { 
						STA = { num_of_controlled_states < 1 }
						SND = { num_of_controlled_states < 1 }
						}
					NOT = {
						AND = { 
							STA = { num_of_controlled_states < 1 }
							SND = { num_of_controlled_states < 1 }	
							}
						}
					}
				clr_country_flag = ziobro_start_war
				set_country_flag = ziobro_start_war_victory
				every_country = {
					limit = { NOT = { is_ai = yes } }
					news_event = { id = ziobro.1 }
					}
				every_country = {
					limit = { has_war_with = PRK }
					white_peace = PRK
					}					
				PRK = {
					end_puppet = OSW
					end_puppet = BOH
				}
				OSW = { drop_cosmetic_tag = yes }				
			}
			else = {		
				clr_country_flag = ziobro_start_war
				set_country_flag = ziobro_start_war_lose
				every_country = {
					limit = { NOT = { is_ai = yes } }
					news_event = { id = ziobro.2 }
					}
				ROOT = {
					white_peace = SND
					white_peace = STA
					end_puppet = OSW
					end_puppet = BOH
				}
				BOH = { drop_cosmetic_tag = yes }
				OSW = { drop_cosmetic_tag = yes }
			}
		}
	}
	


}
