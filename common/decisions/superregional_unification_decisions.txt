western_poland_unification = {

	icon = GFX_decision_icon_default
	
	available = {
		custom_trigger_tooltip = {
			tooltip = controls_western_poland
			can_unify_western_poland = yes
		}
	}
	
	fire_only_once = yes

	cost = 0
	
	visible = { 
		AND = {
			is_western_poland_unifier = yes
			OR = {
				has_global_flag = lodz_and_kalisz_unified
				has_global_flag = pomorze_unified
				has_global_flag = kujawsko_wielkopolskie_unified
			}
			}
		}

	complete_effect = {
		log = "[GetDateText]: Decision westpol_supregional_unification"
		set_country_flag = unified_supregional
		set_global_flag = westpol_unified
		superregional_unification_effect = yes
	}
	ai_will_do = { factor = 1 }
}
