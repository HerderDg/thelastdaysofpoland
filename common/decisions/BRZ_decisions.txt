BRZ_the_dream_team = {
	edit_team = {
		visible = {
			NOT = {
				has_country_flag = team_editor_shown
			}
		}
		complete_effect = {
			set_country_flag = team_editor_shown
			if = {
				limit = {
					NOT = {
						has_variable = current_team_setup
					}
				}
				set_variable = {
					current_team_setup = 442
				}
				hidden_effect = {
					set_initial_team = yes
				}
			}
		}
	}
	debug_team = {
		visible = {
			is_debug = yes
		}
		complete_effect = {
			every_character = {
				limit = {
					has_character_flag = is_active
				}
				clr_character_flag = is_active
			}
		}
	}
}
