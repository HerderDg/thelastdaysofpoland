coring_category = {
	core_the_state = {
		state_target = yes
		target_root_trigger = {
			always = yes
		}
		target_array = ROOT.owned_states
		target_trigger = {
			FROM = {
				NOT = { is_core_of = ROOT }
			}
		}
		available = {
			has_political_power > FROM.state_population_k
        }
		days_remove = 30
		custom_cost_trigger = {
			has_political_power > FROM.state_population_k
		}
		custom_cost_text = coring_cost
		#cost = FROM.state_population_k
		complete_effect = {
			set_temp_variable = { coring_pp_cost = -1 }
			multiply_temp_variable = { coring_pp_cost = FROM.state_population_k }
			add_political_power = coring_pp_cost
		}
		remove_effect = {
			FROM = {
				add_core_of = ROOT
			}
		}
	}
}
