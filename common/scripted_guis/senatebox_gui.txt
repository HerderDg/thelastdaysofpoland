scripted_gui = {
	senatebox_gui = {
		window_name = senatebox_window
		context_type = player_context
		visible = {
			has_country_flag = senatebox_shown
		}
		effects = {
			add_popular_click = {
				add_to_variable = {
					CAL_populares_count = 1
				}
				CAL_redraw_senate_diagram = yes
			}
			subtract_popular_click = {
				subtract_from_variable = {
					CAL_populares_count = 1
				}
				CAL_redraw_senate_diagram = yes
			}
			add_senate_click = {
				add_to_variable = {
					CAL_senate_lvl = 1
				}
				CAL_redraw_senate_diagram = yes
			}
			subtract_senate_click = {
				subtract_from_variable = {
					CAL_senate_lvl = 1
				}
				if = {
					limit = {
						check_variable = {
							CAL_senate_lvl = 3
						}
						check_variable = {
							CAL_populares_count > 300
						}
					}
					set_variable = {
						CAL_populares_count = 300
					}
				}
				else_if = {
					limit = {
						check_variable = {
							CAL_senate_lvl = 2
						}
						check_variable = {
							CAL_populares_count > 150
						}
					}
					set_variable = {
						CAL_populares_count = 150
					}
				}
				else_if = {
					limit = {
						check_variable = {
							CAL_senate_lvl = 1
						}
						check_variable = {
							CAL_populares_count > 100
						}
					}
					set_variable = {
						CAL_populares_count = 100
					}
				}
				CAL_redraw_senate_diagram = yes
			}
		}
		triggers = {
			add_senate_click_enabled = {
				check_variable = {
					CAL_senate_lvl < 4
				}
			}
			subtract_senate_click_enabled = {
				check_variable = {
					CAL_senate_lvl > 1
				}
			}
			add_popular_click_enabled = {
				if = {
					limit = {
						check_variable = {
							CAL_senate_lvl = 1
						}
					}
					check_variable = {
						CAL_populares_count < 100
					}
				}
				else_if = {
					limit = {
						check_variable = {
							CAL_senate_lvl = 2
						}
					}
					check_variable = {
						CAL_populares_count < 150
					}
				}
				else_if = {
					limit = {
						check_variable = {
							CAL_senate_lvl = 3
						}
					}
					check_variable = {
						CAL_populares_count < 300
					}
				}
				else = {
					check_variable = {
						CAL_populares_count < 900
					}
				}
			}
			subtract_popular_click_enabled = {
				check_variable = {
					CAL_populares_count > 0
				}
			}
		}
	}	
}
