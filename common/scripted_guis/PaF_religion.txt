scripted_gui = {

	open_religion_gui = {
		context_type = player_context
		parent_window_name = "political_idea_entry"
		window_name = "religion_button_container"
		effects = {
			show_religion_gui_click = {
				
				if = {
					limit = { has_variable = global.PaF_ReligionGUI }
					clear_variable = global.PaF_ReligionGUI
				}
				else = {
					set_variable = { global.PaF_ReligionGUI = 1 }
				}	
			}
		}
	}

	religion_gui = {
		window_name = "religion_container_gui"
		context_type = player_context
		visible = {
			has_variable = global.PaF_ReligionGUI
		}
		properties = {
			religious_law_value = { 
				image = "[GetCurrentReligionLawIcon]"
			}
		}
	}
}
