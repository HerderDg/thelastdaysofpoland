defined_text = {
	name = GetFourDigitsNumber
	text = {
		trigger = {
			check_variable = {
				caseNo < 10
			}
		}
		localization_key = "000[?caseNo]"
	}
	text = {
		trigger = {
			check_variable = {
				caseNo > 9
			}
			check_variable = {
				caseNo < 100
			}
		}
		localization_key = "00[?caseNo]"
	}
	text = {
		trigger = {
			check_variable = {
				caseNo > 99
			}
			check_variable = {
				caseNo < 1000
			}
		}
		localization_key = "0[?caseNo]"
	}
	text = {
		localization_key = "[?caseNo]"
	}
}
