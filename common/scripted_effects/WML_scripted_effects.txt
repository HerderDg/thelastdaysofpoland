WML_calculate_action_cost_ZYD = {
	if = {
		limit = {
			check_variable = {
				dialogue_ZYD < 25
			}
		}
		set_variable = {
			action_cost = 20
		}
	}
	else_if = {
		limit = {
			check_variable = {
				dialogue_ZYD < 50
			}
		}
		set_variable = {
			action_cost = 15
		}
	}
	else_if = {
		limit = {
			check_variable = {
				dialogue_ZYD < 75
			}
		}
		set_variable = {
			action_cost = 10
		}
	}
	else = {
		set_variable = {
			action_cost = 5
		}
	}
	if = {
		limit = {
			has_variable = dialogue_ZYD_multiplier
		}
		multiply_variable = {
			action_cost = dialogue_ZYD_multiplier
		}
	}
	log = "Action cost: [?var:action_cost]"
}
