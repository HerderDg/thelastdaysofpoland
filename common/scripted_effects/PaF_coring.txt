coring_effect = {
	set_temp_variable = { coring_pp_cost = THIS.coring_cost }
	set_temp_variable = { coring_multiplier = -1}
	subtract_from_temp_variable = {	coring_multiplier = modifier@coring_cost_factor}
	multiply_temp_variable = { coring_pp_cost = coring_multiplier }
	THIS.owner = {
		add_political_power = coring_pp_cost
	}
	THIS = {
		PaF_change_capital_coring_prequel = yes
	}
	add_core_of = ROOT
	remove_from_array = { ROOT.noncores_array = THIS.id }
	set_temp_variable = { coreable_temp = 0}
	ROOT = {
		every_owned_state = {
			limit = {
				NOT = { is_core_of = ROOT }
				compliance > 75
				check_variable = {
					THIS.coring_cost < ROOT.political_power
				}
			}
			add_to_temp_variable = { coreable_temp = 1}
		}
		set_variable = {coreable_number = coreable_temp }
	}
}

}

	set_global_flag = coring_ui_visible
}

close_coring_ui_effect = {

}
