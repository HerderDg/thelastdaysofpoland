PRK_revolt_NRA = {
	hidden_effect = {
		if = {
			limit = {
				has_country_flag = PRK_ziobro_lost
			}
			NRA = {
				add_state_core = 860
				add_state_core = 679
			}
		}
		else = {
			NRA = {
				add_state_core = 860
				add_state_core = 378
			}
		}
		PRK = { 
			add_popularity = {
				ideology = marxism_cranism
				popularity = -0.16
			}
		}
		every_owned_state = {
			limit = { is_core_of = NRA }
			NRA = { transfer_state = PREV }
			PRK = { 
				remove_state_core = PREV
				add_state_claim = PREV
			}
		}
		NRA = {
			load_oob = "NRA_2027"
			declare_war_on = {
				target = PRK
				type = annex_everything
			}
			delete_unit_template_and_units = {
				division_template = "Batalion Piechoty"
				disband = yes
			}
			delete_unit_template_and_units = {
				division_template = "Milicja Prokuratorska"
				disband = yes
			}
			delete_unit_template_and_units = {
				division_template = "Straż Obywatelska"
				disband = yes
			}
		}
		every_country = {
			limit = { NOT = { is_ai = yes } }
			news_event = { id = ziobro.3 }
		}
	}
	log = "[GetDateText]: PRK_revolt_NRA"
}
