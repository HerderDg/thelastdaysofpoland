
MNI_collapse_OPO = {
	hidden_effect = {
		OPO = {
			add_state_core = 428
			add_state_core = 425
		}

		MNI = {

		}

			limit = { is_core_of = OPO }
			OPO = { transfer_state = PREV }
		}
	}
	log = "[GetDateText]: MNI_collapse_OPO"
}
