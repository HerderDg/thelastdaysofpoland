country_event = {
	id = bedzincon.1 #congress of bedzin
	title = bedzincon.1.t
	desc = bedzincon.1.desc

	fire_only_once = yes
	is_triggered_only = yes
	
	immediate = { 
		log = "[GetDateText]: [ROOT.GetName] fires bedzincon.1"
		set_variable = {
			BEDZINCON_SOS = 50
		}
		set_variable = {
			BEDZINCON_DGR = 30
		}
		set_variable = {
			BEDZINCON_JAW = 20
		}				
	}
	
	option = {
		name = bedzincon.1.a
		set_global_flag = ZAG_congress_of_bedzin
		if = { limit = { JAW = { is_ai = no } check_variable = { BEDZINCON_JAW > 1 } } JAW = { country_event = bedzincon.2 } }
		if = { limit = { JAW = { is_ai = no } check_variable = { BEDZINCON_DGR > 1 } } DGR = { country_event = bedzincon.2 } }
		if = { limit = { JAW = { is_ai = no } check_variable = { BEDZINCON_SOS > 1 } } SOS = { country_event = bedzincon.2 } }
		if = {
			limit = {
					JAW = { is_ai = yes }
					DGR = { is_ai = yes }
					SOS = { is_ai = yes }
			}
			SOS = { country_event = bedzincon.2 }
		}			
	}
}
