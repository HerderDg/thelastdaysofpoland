country_event = {
	id = honor.1 #rewolucja w durś
	title = honor.1.t
	desc = honor.1.desc

	fire_only_once = yes
	is_triggered_only = yes

	immediate = { 

		log = "[GetDateText]: [ROOT.GetName] fires honor.1" 
	}

	option = {
		name = honor.1.a
		DUR = { drop_cosmetic_tag = yes }
		MNI = {
		diplomatic_relation = {
				country = DUR
				relation = puppet
				active = no
			}
		}
		GER = { remove_from_faction = DUR }
		country_event = {
			id = honor.2
			days = 13
		}
	}
}

	desc = honor.2.desc

	fire_only_once = yes
	is_triggered_only = yes

	immediate = { 
		DUR_collapse_WOD = yes 
		log = "[GetDateText]: [ROOT.GetName] fires honor.2" 
		DUR = {
			country_event = {
				id = honor.3
				days = 10
			}
		}
	}

	option = {
		name = honor.2.a
		DUR = { 
			declare_war_on = {
				target = WOD
				type = voivodeship_war
			}
		}
	}
	option = {
		name = honor.2.b
		DUR = {
			clear_compatible_countries = yes
		}
		WOD = {
			change_tag_from = DUR
			if = {
				limit = {
					WOD = {
						is_ai = no
					}
				}
				set_variable = { global.PaF_Opening_Screen = 1 }
			}
		}
		DUR = { 
			declare_war_on = {
				target = WOD
				type = voivodeship_war
			}
		}
	}
}
