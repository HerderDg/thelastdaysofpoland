country_event = {
	id = buzzer.4 #mecz ze krzepicami
	title = { 
		trigger = {
			has_country_flag = krzepice_kickoff_victory
		}
		text = buzzer.4.t.w
	}
	title = { 
		text = buzzer.4.t
	}
	desc = {
		trigger = {
			has_country_flag = krzepice_kickoff_victory
		}
		text = buzzer.4.desc.w
	}
	desc = { 
		text = buzzer.4.desc
	}
	fire_only_once = yes
	is_triggered_only = yes
	immediate = { log = "[GetDateText]: [ROOT.GetName] fires buzzer.4"}
	option = {

		name = buzzer.4.a

		trigger = {
			has_country_flag = krzepice_kickoff_victory
		}
		
	}
	option = {

		name = buzzer.4.b

		trigger = {
			not = {
				has_country_flag = krzepice_kickoff_victory
			}
		}
		
	}
}
