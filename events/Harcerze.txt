	id = zhp.16 #wojna wśród harcerzy
	title = zhp.16.t
	desc = zhp.16.desc

	immediate = { 
		log = "[GetDateText]: [ROOT.GetName] fires zhp.16"
		hidden_effect = {
			damage_units = {
				damage = 0.50
			}
		}
	}

	fire_only_once = yes
	trigger = {
		original_tag = ZHP
		has_war = no
		can_unify_lodz = yes
		}
	mean_time_to_happen = {
		months = 2
	}
	option = {
		name = zhp.16.a # idealiści
		trigger = {
			#has_country_flag = idealist_takeover
			has_country_flag = ZHP_ZHP_Path_2
		}
		hidden_effect = {
			set_nationality = {
				target_country = ZHR
				character = ZHP_marek_jedraszewski
			}
		}
		set_politics = {
			ruling_party = postpolonism
		}
		set_country_leader_ideology = scout_system_subtype
		ZHP_add_states_to_rebels = yes
		for_each_loop = {
			array = ZHP_breaker_states
			
			ZHR = {
				transfer_state = v
				add_state_core = v
			}
		}
		ZHR = {
			load_oob = "ZHR_ins"			
			declare_war_on = {
				target = ZHP
				type = annex_everything
			}
		}
	}
	option = {
		name = zhp.16.b #konserwatyści
			trigger = {
				#has_country_flag = conservative_takeover
				has_country_flag = ZHP_ZHR_Path_2
			}
		hidden_effect = {
			set_nationality = {
				target_country = GFL
				character = ZHP_orsza
			}
		}
		set_politics = {
			ruling_party = autocracy
		}
		set_country_leader_ideology = autocracy_subtype
		ZHP_add_states_to_rebels = yes
		for_each_loop = {
			array = ZHP_breaker_states
			GFL = {
				transfer_state = v
				add_state_core = v
			}
		}
		
		GFL = {
			load_oob = "GFL_ins"
			declare_war_on = {
				target = ZHP
				type = annex_everything
			}
		}
	}
}
