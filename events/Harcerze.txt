	id = zhp.16 #wojna wśród harcerzy; todo: opóźnienie wystrzelenia eventu; podzielenie ZHP
	title = zhp.16.t
	desc = zhp.16.desc

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = zhp.16.a # idealiści
		trigger = {
			or = {
				if = {
					limit = {
						has_variable = zhp_idealist_influence
						has_variable = zhp_conservative_influence
					}
					check_variable = { zhp_idealist_influence < zhp_conservative_influence}
				}
				else = {
					has_variable = zhp_idealist_influence
				}
			}
		}
	}
	option = {
		name = zhp.16.b #konserwatyści
		trigger = {
			or = {
				if = {
					limit = {
						has_variable = zhp_idealist_influence
						has_variable = zhp_conservative_influence
					}
					check_variable = { zhp_idealist_influence > zhp_conservative_influence}
				}
				else = {
					has_variable = zhp_conservative_influence
				}
			}
		}
	}
}
