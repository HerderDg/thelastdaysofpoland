	id = kszrozpad.4
	title = kszrozpad.4.t
	desc = kszrozpad.4.desc

	fire_only_once = yes
	is_triggered_only = yes

	immediate = {
		log = "[GetDateText]: [ROOT.GetName] fires kszrozpad.4"
		hidden_effect = {
			white_peace = HEL
			white_peace = OWS
			white_peace = KSC
			white_peace = WHR
			KSZ_collapse_HEL = yes
			KSZ_collapse_OWS = yes
			KSZ_collapse_KSC = yes
			WHR = {
				add_state_core = 1075
				transfer_state = 1075

				set_capital = { state = 1075 }

				remove_ideas = WHR_grand_griffin_uprising
				add_ideas = WHR_hostile_city
			}
		}
	}

	option = {
		name = kszrozpad.4.a
		every_state = {
			limit = {
				is_pomerelia_state = yes
			}
			teleport_armies = {
				limit = {
					original_tag = KSZ
					is_ai = yes
				}
			}
		}
		remove_ideas = {
			KSZ_griffin_strikes
		}
		swap_ideas = {
			remove_idea = KSZ_kashubians_kocievians_borovians
			add_idea = KSZ_to_defend_kashubian_freedom
		}
		add_ideas = {
			KSZ_aftershocks_of_anarchy_and_insurrection
			KSZ_state_in_a_limbo
		}
	}
