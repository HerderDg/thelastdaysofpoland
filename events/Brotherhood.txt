				producer = ZYD
				}

		 limit = { has_tech = ranged_weapons1 }

				}
			}

		 	 	add_equipment_to_stockpile = {
				type = ranged_equipment_0
				amount = 1000

				}
			}
		}

			add_equipment_to_stockpile = {
				type = melee_equipment_2
				amount = 1000
				producer = ZYD
				}
		 else_if = {
